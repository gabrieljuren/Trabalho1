        public int Registro(int id)

        {
            int contador = 0;

            try
            {
                cmd = new SqlCommand("Select * from WebServiceTest where Id = " + id + "", cn);
                dr = cmd.ExecuteReader();
                while (dr.Read())
                {
                    contador++;
                }

                dr.Close();
            }

            catch (Exception ex)
            {
                MessageBox.Show("Não foi bem executado: " + ex.ToString());

            }

            return contador;
        }
